Ejemplo de ventana de diálogo (modal) HTML, CSS y JS

Confirmación/Question

                
                    const ShowConfirm = async () => {
                        resp = await DialogForm.ShowQuestionAsync({
                                    Titulo: "Guardar información",
                                    Mensaje: "¿La información es correcta?"
                                    })
                        if (resp == "OK") {
                            console.log("Se acepta continuar");
                            }
                        else
                            {
                            console.log("Se canceló el proceso");
                        }
                    }

Aviso/Alert

                
                    const ShowAlert = async () => {
                        resp = await DialogForm.ShowInformationAsync({ 
                                    Titulo: "Alta de prospecto", 
                                    Mensaje: "El prospecto se agrego con éxito" 
                                    })
                        console.log("Se leyó el aviso");            
                    }

Procesando/Working

                
                    const ShowProcess = async () => {
                        DialogForm.ShowWaiting({ 
                            Titulo: "Guardando información", 
                            Mensaje: "Por favor espera..." })  
                        setTimeout(() => { DialogForm.Close() }, 5000)            
                    }
